1) INSTALAR XAMMP
2) INGRESAR A localhost/phpmyadmin/
3) CREAR DB "appchat" utf8_spanish2_ci
4) CREAR TABLA "chat"
	TABLA:
		 id` int(11) NOT NULL, autoincrement primary key
		 `nombre` varchar(255) 
		 `mensaje` varchar(255)
		 `fecha` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP

5) LINK localhost/chat/
